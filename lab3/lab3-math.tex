\section*{Equilibrium}

\[
\text{At equilibrium: } \sum F = 0 \implies mg = k_{\text{eff}} \Delta x
\]
\[
\Delta x = \frac{g}{k_{\text{eff}}} m \qquad (\text{linear: slope} = g/k_{\text{eff}})
\]
\[
k_{\text{eff}} = \frac{g}{\text{slope}}
\]

\section*{Simple Harmonic Motion}

\[
ma = -k_{\text{eff}} x \implies \frac{d^2 x}{dt^2} + \frac{k_{\text{eff}}}{m} x = 0
\]
\[
\omega = \sqrt{k_{\text{eff}}/m}, \quad T = 2\pi/\omega = 2\pi\sqrt{m/k_{\text{eff}}}
\]
\[
T^2 = \frac{4\pi^2}{k_{\text{eff}}} m \qquad (\text{linear: slope} = 4\pi^2/k_{\text{eff}})
\]
\[
k_{\text{eff}} = \frac{4\pi^2}{\text{slope}}
\]

\section*{Parallel Springs}

Two springs $k_1, k_2$ in parallel: same displacement $\Delta x$.
\[
F_{\text{total}} = F_1 + F_2 = k_1 \Delta x + k_2 \Delta x
\]
\[
k_{\text{eff}} \Delta x = (k_1 + k_2) \Delta x
\]
\[
\implies k_{\text{eff}} = k_1 + k_2 \qquad \text{(Binary Formula 1)}
\]

\section*{Series Springs}

Two springs $k_1, k_2$ in series: same force $F$.
\[
\Delta x_{\text{total}} = \Delta x_1 + \Delta x_2 = \frac{F}{k_1} + \frac{F}{k_2}
\]
\[
\frac{F}{k_{\text{eff}}} = F\left(\frac{1}{k_1} + \frac{1}{k_2}\right)
\]
\[
\implies \frac{1}{k_{\text{eff}}} = \frac{1}{k_1} + \frac{1}{k_2}
\]
\[
\implies k_{\text{eff}} = \frac{k_1 k_2}{k_1 + k_2} \qquad \text{(Binary Formula 2)}
\]
